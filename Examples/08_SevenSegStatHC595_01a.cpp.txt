{
	gpioPinId_t myDspPins[]{{GPIOA, 1 << 5}, {GPIOA, 1 << 6}, {GPIOB, 1 << 12}};
	uint8_t tmpDispData[]{0x91, 0x83, 0xA0, 0xC2};	//The display buffer loaded with "YbaG"

	myDspHw.setDspBuffPtr(tmpDispData);
	myDspHw.begin();

	for(;;)

		vTaskDelay(1);
	}
